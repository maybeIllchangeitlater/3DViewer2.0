#version 410 core

in vec3 position;
in vec2 texture;
in vec3 normal;

out vec3 fragNormal;
out vec2 textureCoord;
out vec3 FragPosistion;

uniform mat4 projectionMatrix;
uniform mat4 modelViewMatrix;

void main()
{
    fragPosistion = vec3(modelViewMatrix * vec4(position, 1.0));
    fragNormal = mat3(transpose(inverse(modelViewMatrix))) * normalize(normal);
    textureCoord = vec2(texture.x, 1.0 - texture.y);
    gl_Position = projectionMatrix * vec4(fragPosistion, 1.0);
}
