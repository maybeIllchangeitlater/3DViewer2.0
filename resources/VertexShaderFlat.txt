#version 410 core

in vec3 position;
in vec3 normal;

out vec3 fragNormal;

uniform mat4 projectionMatrix;
uniform mat4 modelViewMatrix;

void main()
{
    gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
    fragNormal = mat3(transpose(inverse(modelViewMatrix))) * normalize(normal);
}
