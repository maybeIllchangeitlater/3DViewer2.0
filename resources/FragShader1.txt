uniform vec4 color;
uniform bool dashed;
     void main()
     {
         if (dashed)
         {
             float dashSize = 0.3;
             float gapSize = 0.3;
             float t = mod(gl_FragCoord.x, (dashSize + gapSize)) / (dashSize
     + gapSize);
             if (t > 0.5)
             {
                 gl_FragColor = color;
             }
             else
             {
                 discard;
             }
         }
         else
         {
             gl_FragColor = color;
         }
     }
